 72def module_to_kore(definition: KDefinition) -> Module:
 73    """Convert the main module of a kompiled KAST definition to KORE format."""
 74
 75    module = simplified_module(definition)
 76
 77    name = name_to_kore(module.name)
 78    attrs = atts_to_kore({key: value for key, value in module.att.items() if key != 'digest'})  # filter digest
 79
 80    imports = [Import('K')]
 81    sort_decls = [
 82        sort_decl_to_kore(syntax_sort)
 83        for syntax_sort in module.syntax_sorts
 84        if syntax_sort.sort.name not in [K.name, K_ITEM.name]
 85    ]
 86
 87    sentences: list[Sentence] = []
 88    sentences += imports
 89    sentences += sort_decls
 90
 91    return Module(name=name, sentences=sentences, attrs=attrs)

101def atts_to_kore(att: Mapping[str, Any]) -> list[App]:
102    res = [att_to_kore(key, value) for key, value in att.items()]
103    res.sort(key=lambda app: app.symbol)
104    return res

107def att_to_kore(key: str, value: Any) -> App:
108    symbol = name_to_kore(key)
109
110    if value == '':
111        return App(symbol)
112
113    parse_res = _parse_special_att_value(key, value)
114    if parse_res is not None:
115        sorts, args = parse_res
116        return App(symbol, sorts, args)
117
118    if isinstance(value, str):
119        return App(symbol, (), (String(value),))
120
121    if isinstance(value, FrozenDict) and 'node' in value:
122        # TODO Should be kast_to_kore, but we do not have a KompiledKore.
123        # TODO We should be able to add injections based on info in KDefinition.
124        pattern = _kast_to_kore(KInner.from_dict(value))
125        if not isinstance(pattern, App):
126            raise ValueError('Expected application as attribure, got: {pattern.text}')
127        return App(symbol, (), (pattern,))
128
129    raise ValueError(f'Attribute conversion is not implemented for: {key}: {value}')

132def _parse_special_att_value(key: str, value: Any) -> tuple[tuple[Sort, ...], tuple[Pattern, ...]] | None:
133    if key == KAtt.LOCATION:
134        assert isinstance(value, tuple)
135        assert len(value) == 4
136        loc_str = ','.join(str(loc) for loc in value)
137        return (), (String(f'Location({loc_str})'),)
138    if key == KAtt.SOURCE:
139        assert isinstance(value, str)
140        return (), (String(f'Source({value})'),)
141    return None

165def simplified_module(definition: KDefinition, module_name: str | None = None) -> KFlatModule:
166    """
167    In ModuleToKORE.java, there are some implicit KAST-to-KAST kompilation
168    steps hidden in the conversion. In particular, the kompiled KAST definition
169    (compiled.json) is modular, whereas the kompiled definition
170    (definition.kore) is flat.
171
172    This function aims to factor out these hidden KAST-to-KAST kompilation
173    steps so that our implementation of module_to_kore can be as simple as
174    possible. Moreover, this has the potential to shed some light on how
175    modules can be kompiled incrementally.
176
177    This function is an approximation, i.e. there might be cases where it
178    produces a different result to what would be expected based on kompile's
179    output. These discrepancies should be analyzed and fixed.
180    """
181    module_name = module_name or definition.main_module_name
182    module = _flatten_module(definition, module_name)  # TODO KORE supports imports, why is definition.kore flat?
183    module = _add_syntax_sorts(module)
184    module = _add_collection_atts(module)
185    module = _add_domain_value_atts(module)
186    return module

189def _flatten_module(definition: KDefinition, module_name: str) -> KFlatModule:
190    """Return a flat module with all sentences included and without imports"""
191    module = definition.module(module_name)
192    sentences = _imported_sentences(definition, module_name)
193    return module.let(sentences=sentences, imports=())

196def _imported_sentences(definition: KDefinition, module_name: str) -> list[KSentence]:
197    """Return all sentences from imported modules, including the module itself."""
198
199    pending: list[str] = [module_name]
200    imported: set[str] = set()
201
202    res: list[KSentence] = []
203    while pending:
204        module_name = pending.pop()
205        if module_name in imported:
206            continue
207        module = definition.module(module_name)
208        res += module.sentences
209        pending += (importt.name for importt in module.imports)
210        imported.add(module_name)
211
212    return res

215def _add_syntax_sorts(module: KFlatModule) -> KFlatModule:
216    """Return a module with explicit syntax declarations: each sort is declared with the union of its attributes"""
217    sentences = [sentence for sentence in module if not isinstance(sentence, KSyntaxSort)]
218    sentences += _syntax_sorts(module)
219    return module.let(sentences=sentences)

221
222def _syntax_sorts(module: KFlatModule) -> list[KSyntaxSort]:
223    """Return a declaration for each sort in the module"""
224    declarations: dict[KSort, KAtt] = {}
225
226    def is_higher_order(production: KProduction) -> bool:
227        # Example: syntax {Sort} Sort ::= Sort "#as" Sort
228        return production.sort in production.params
229
230    # Merge attributes from KSyntaxSort instances
231    for syntax_sort in module.syntax_sorts:
232        sort = syntax_sort.sort
233        if sort not in declarations:
234            declarations[sort] = syntax_sort.att
235        else:
236            assert declarations[sort].keys().isdisjoint(syntax_sort.att)
237            declarations[sort] = declarations[sort].update(syntax_sort.att)
238
239    # Also consider production sorts
240    for production in module.productions:
241        if is_higher_order(production):
242            continue
243
244        sort = production.sort
245        if sort not in declarations:
246            declarations[sort] = EMPTY_ATT
247
248    return [KSyntaxSort(sort, att=att) for sort, att in declarations.items()]

251def _add_collection_atts(module: KFlatModule) -> KFlatModule:
252    """Return a module where concat, element and unit attributes are added to collection sort declarations"""
253
254    # Example: syntax Map ::= Map Map [..., klabel(_Map_), element(_|->_), unit(.Map), ...]
255    concat_prods = {prod.sort: prod for prod in module.productions if KAtt.ELEMENT in prod.att}
256
257    assert all(
258        KAtt.UNIT in prod.att for _, prod in concat_prods.items()
259    )  # TODO Could be saved with a different attribute structure: concat(Element, Unit)
260
261    def update_att(sentence: KSentence) -> KSentence:
262        if not isinstance(sentence, KSyntaxSort):
263            return sentence
264
265        syntax_sort: KSyntaxSort = sentence
266
267        if syntax_sort.att.get(KAtt.HOOK) not in COLLECTION_HOOKS:
268            return syntax_sort
269
270        prod_att = concat_prods[syntax_sort.sort].att
271
272        # TODO Here, attriubtes are stored as dict, but ultimately we should parse known attributes in KAtt.from_dict
273        return syntax_sort.let(
274            att=syntax_sort.att.update(
275                {
276                    KAtt.CONCAT: KApply(prod_att[KAtt.KLABEL]).to_dict(),
277                    KAtt.ELEMENT: KApply(prod_att[KAtt.ELEMENT]).to_dict(),
278                    KAtt.UNIT: KApply(prod_att[KAtt.UNIT]).to_dict(),
279                }
280            )
281        )
282
283    sentences = tuple(update_att(sent) for sent in module)
284    return module.let(sentences=sentences)

287def _add_domain_value_atts(module: KFlatModule) -> KFlatModule:
288    """
289    Return a module where attribute "hasDomainValues" is added to all
290    sort declarations that either have the "token" attribute directly,
291    or on a corresponding production.
292    """
293
294    token_sorts = _token_sorts(module)
295
296    def update_att(sentence: KSentence) -> KSentence:
297        if not isinstance(sentence, KSyntaxSort):
298            return sentence
299
300        syntax_sort: KSyntaxSort = sentence
301
302        if syntax_sort.sort not in token_sorts:
303            return syntax_sort
304
305        return syntax_sort.let(att=syntax_sort.att.update({KAtt.HAS_DOMAIN_VALUES: ''}))
306
307    sentences = tuple(update_att(sent) for sent in module)
308    return module.let(sentences=sentences)
309

311def _token_sorts(module: KFlatModule) -> set[KSort]:
312    res: set[KSort] = set()
313
314    # TODO "token" should be an attribute of only productions
315    for syntax_sort in module.syntax_sorts:
316        if KAtt.TOKEN in syntax_sort.att:
317            res.add(syntax_sort.sort)
318
319    for production in module.productions:
320        if KAtt.TOKEN in production.att:
321            res.add(production.sort)
322
323    return res

359def kast_to_kore(
360    kast_defn: KDefinition,
361    kompiled_kore: KompiledKore,
362    kast: KInner,
363    sort: KSort | None = None,
364) -> Pattern:
365    if sort is None:
366        sort = K
367    kast = kast_defn.add_ksequence_under_k_productions(kast)
368    kast = kast_defn.sort_vars(kast, sort)
369    kast = kast_defn.add_cell_map_items(kast)
370    kast = kast_defn.add_sort_params(kast)
371    kore = _kast_to_kore(kast)
372    kore = kompiled_kore.add_injections(kore, _ksort_to_kore(sort))
373    return kore

377def krule_to_kore(kast_defn: KDefinition, kompiled_kore: KompiledKore, krule: KRule) -> Axiom:
378    krule_body = krule.body
379    krule_lhs = CTerm(extract_lhs(krule_body), [bool_to_ml_pred(krule.requires)])
380    krule_rhs = CTerm(extract_rhs(krule_body), [bool_to_ml_pred(krule.ensures)])
381
382    top_level_kore_sort = SortApp('SortGeneratedTopCell')
383    top_level_k_sort = KSort('GeneratedTopCell')
384    # The backend does not like rewrite rules without a precondition
385    if len(krule_lhs.constraints) > 0:
386        kore_lhs0: Pattern = kast_to_kore(kast_defn, kompiled_kore, krule_lhs.kast, sort=top_level_k_sort)
387    else:
388        kore_lhs0 = And(
389            top_level_kore_sort,
390            (
391                kast_to_kore(kast_defn, kompiled_kore, krule_lhs.kast, sort=top_level_k_sort),
392                Top(top_level_kore_sort),
393            ),
394        )
395
396    kore_rhs0: Pattern = kast_to_kore(kast_defn, kompiled_kore, krule_rhs.kast, sort=top_level_k_sort)
397
398    kore_lhs = kompiled_kore.add_injections(kore_lhs0, sort=top_level_kore_sort)
399    kore_rhs = kompiled_kore.add_injections(kore_rhs0, sort=top_level_kore_sort)
400    prio = krule.priority
401    attrs = [App(symbol='priority', sorts=(), args=(String(str(prio)),))]
402    if 'label' in krule.att:
403        label = krule.att['label']
404        attrs.append(App(symbol='label', sorts=(), args=(String(label),)))
405    axiom = Axiom(
406        vars=(),
407        pattern=Rewrites(
408            sort=top_level_kore_sort,
409            left=kore_lhs,
410            right=kore_rhs,
411        ),
412        attrs=attrs,
413    )
414    return axiom

417def kflatmodule_to_kore(kast_defn: KDefinition, kompiled_kore: KompiledKore, kflatmodule: KFlatModule) -> Module:
418    kore_axioms: list[Sentence] = []
419    for sent in kflatmodule.sentences:
420        if type(sent) is not KRule:
421            raise ValueError(f'Cannot convert sentence to Kore: {sent}')
422        kore_axioms.append(krule_to_kore(kast_defn, kompiled_kore, sent))
423    imports: list[Sentence] = [_kimport_to_kore(kimport) for kimport in kflatmodule.imports]
424    return Module(name=kflatmodule.name, sentences=(imports + kore_axioms))

435def _kast_to_kore(term: KInner) -> Pattern:
436    stack: list = [term, []]
437    while True:
438        patterns = stack[-1]
439        term = stack[-2]
440        idx = len(patterns) - len(term.terms)
441        if not idx:
442            stack.pop()
443            stack.pop()
444            pattern = _kinner_to_kore(term, patterns)
445            if not stack:
446                return pattern
447            stack[-1].append(pattern)
448        else:
449            stack.append(term.terms[idx])
450            stack.append([])

453def _kinner_to_kore(kinner: KInner, patterns: list[Pattern]) -> Pattern:
454    match kinner:
455        case KToken():
456            assert not patterns
457            return _ktoken_to_kore(kinner)
458        case KVariable():
459            assert not patterns
460            return _kvariable_to_kore(kinner)
461        case KSequence():
462            return _ksequence_to_kore(kinner, patterns)
463        case KApply():
464            return _kapply_to_kore(kinner, patterns)
465        case _:
466            raise ValueError(f'Unsupported KInner: {kinner}')

469def _ktoken_to_kore(ktoken: KToken) -> DV:
470    value: String
471    if ktoken.sort == STRING:
472        value = String(pretty_string(ktoken))
473    elif ktoken.sort == BYTES:
474        value = String(pretty_bytes_str(ktoken))

476        value = String(ktoken.token)

479
480    return DV(sort, value)

492def _ksequence_to_kore(kseq: KSequence, patterns: list[Pattern]) -> Pattern:
493    if not patterns:
494        return DOTK
495
496    unit: Pattern
497    args: list[Pattern]
498
499    last = patterns[-1]
500    if type(last) is EVar and last.sort == SORT_K:
501        unit = last
502        args = patterns[:-1]
503    else:
504        unit = DOTK
505        args = patterns
506
507    args.reverse()
508    return reduce(lambda x, y: App('kseq', (), (y, x)), args, unit)

553def _kore_to_kast(pattern: Pattern) -> KInner:
554    stack: list = [pattern.pattern if isinstance(pattern, Assoc) else pattern, []]
555    while True:
556        terms = stack[-1]
557        pattern = stack[-2]
558        patterns = pattern.patterns
559        idx = len(terms) - len(patterns)
560        if not idx:
561            stack.pop()
562            stack.pop()
563            term = _pattern_to_kast(pattern, terms)
564            if not stack:
565                return term
566            stack[-1].append(term)
567        else:
568            pattern = patterns[idx]
569            stack.append(pattern.pattern if isinstance(pattern, Assoc) else pattern)
570            stack.append([])
571
572
573def _pattern_to_kast(pattern: Pattern, terms: list[KInner]) -> KInner:
574    match pattern:
575        case DV(sort, String(value)):
576            assert not terms
577            if sort == KORE_STRING:
578                return stringToken(value)
579            if sort == KORE_BYTES:
580                return bytesToken_from_str(value)
581            return KToken(value, _sort_to_kast(sort))
582
583        case EVar(name, sort):
584            assert not terms
585            return KVariable(name=unmunge(name[3:]), sort=_sort_to_kast(sort))
586
587        case App(symbol, sorts, _):
588            if symbol == 'inj':
589                _, _ = sorts
590                (term,) = terms
591                return term
592
593            elif not sorts:
594                if symbol == 'dotk':
595                    () = terms
596                    return KSequence(())
597
598                elif symbol == 'kseq':
599                    _, _ = terms
600                    return KSequence(terms)
601
602                else:
603                    klabel = KLabel(unmunge(symbol[3:]))
604                    return KApply(klabel, terms)
605
606            # hardcoded polymorphic operators
607            elif (
608                symbol
609                == "Lbl'Hash'if'UndsHash'then'UndsHash'else'UndsHash'fi'Unds'K-EQUAL-SYNTAX'Unds'Sort'Unds'Bool'Unds'Sort'Unds'Sort"
610            ):
611                (sort,) = sorts
612                klabel = KLabel(unmunge(symbol[3:]), (_sort_to_kast(sort),))
613                return KApply(klabel, terms)
614
615            else:
616                raise ValueError(f'Unsupported polymorphic operator: {symbol}')
617
618        case Top(sort):
619            assert not terms
620            return mlTop(sort=_sort_to_kast(sort))
621
622        case Bottom(sort):
623            assert not terms
624            return mlBottom(sort=_sort_to_kast(sort))
625
626        case And(sort, _):
627            return mlAnd(terms, sort=_sort_to_kast(sort))
628
629        case Implies(sort, _, _):
630            larg, rarg = terms
631            return mlImplies(larg, rarg, sort=_sort_to_kast(sort))
632
633        case Not(sort, _):
634            (karg,) = terms
635            return mlNot(karg, sort=_sort_to_kast(sort))
636
637        case Exists(sort, EVar(vname, vsort), _):
638            kvar = KVariable(name=unmunge(vname[3:]), sort=_sort_to_kast(vsort))
639            (body,) = terms
640            return mlExists(kvar, body, sort=_sort_to_kast(sort))
641
642        case Equals(op_sort, sort, _, _):
643            larg, rarg = terms
644            return mlEquals(
645                larg,
646                rarg,
647                arg_sort=_sort_to_kast(op_sort),
648                sort=_sort_to_kast(sort),
649            )
650
651        case Ceil(op_sort, sort, _):
652            (karg,) = terms
653            return mlCeil(
654                karg,
655                arg_sort=_sort_to_kast(op_sort),
656                sort=_sort_to_kast(sort),
657            )
658
659        case _:
660            raise ValueError(f'Unsupported Pattern: {pattern}')

714def munge(label: str) -> str:
715    symbol = ''
716    quot = False
717    for c in label:
718        if c in MUNGE_TABLE:
719            symbol += "'" if not quot else ''
720            symbol += MUNGE_TABLE[c]
721            quot = True
722        else:
723            symbol += "'" if quot else ''
724            symbol += c
725            quot = False
726    symbol += "'" if quot else ''
727    return symbol

734def unmunged(it: Iterable[str]) -> Iterator[str]:
735    it = iter(it)
736
737    startquote = False
738    quote = False
739    endquote = False
740    buff: list[str] = []
741    cnt = 0  # len(buff)
742
743    for c in it:
744        if c == "'":
745            if startquote:
746                raise ValueError('Empty quoted section')
747            elif quote:
748                if cnt == 0:
749                    quote = False
750                    endquote = True
751                else:
752                    fragment = ''.join(buff)
753                    raise ValueError(f'Unexpected end of symbol: {fragment}')
754            elif endquote:
755                raise ValueError('Quoted sections next to each other')
756            else:
757                quote = True
758                startquote = True
759
760        else:
761            startquote = False
762            endquote = False
763            if quote:
764                if cnt == 3:
765                    buff.append(c)
766                    munged = ''.join(buff)
767                    buff = []
768                    cnt = 0
769                    if not munged in UNMUNGE_TABLE:
770                        raise ValueError(f'Unknown encoding "{munged}"')
771                    yield UNMUNGE_TABLE[munged]
772                else:
773                    buff += [c]
774                    cnt += 1
775            else:
776                yield c
777
778    if quote:
779        raise ValueError('Unterminated quoted section')
