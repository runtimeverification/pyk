 83def kast_to_kore(
 84    kast_defn: KDefinition,
 85    kompiled_kore: KompiledKore,
 86    kast: KInner,
 87    sort: KSort | None = None,
 88) -> Pattern:
 89    if sort is None:
 90        sort = K
 91    kast = kast_defn.add_ksequence_under_k_productions(kast)
 92    kast = kast_defn.sort_vars(kast, sort)
 93    kast = kast_defn.add_cell_map_items(kast)
 94    kast = kast_defn.add_sort_params(kast)
 95    kore = _kast_to_kore(kast)
 96    kore = kompiled_kore.add_injections(kore, _ksort_to_kore(sort))
 97    return kore

101def krule_to_kore(kast_defn: KDefinition, kompiled_kore: KompiledKore, krule: KRule) -> Axiom:
102    krule_body = krule.body
103    krule_lhs = CTerm(extract_lhs(krule_body), [bool_to_ml_pred(krule.requires)])
104    krule_rhs = CTerm(extract_rhs(krule_body), [bool_to_ml_pred(krule.ensures)])
105
106    top_level_kore_sort = SortApp('SortGeneratedTopCell')
107    top_level_k_sort = KSort('GeneratedTopCell')
108    # The backend does not like rewrite rules without a precondition
109    if len(krule_lhs.constraints) > 0:
110        kore_lhs0: Pattern = kast_to_kore(kast_defn, kompiled_kore, krule_lhs.kast, sort=top_level_k_sort)
111    else:
112        kore_lhs0 = And(
113            top_level_kore_sort,
114            (
115                kast_to_kore(kast_defn, kompiled_kore, krule_lhs.kast, sort=top_level_k_sort),
116                Top(top_level_kore_sort),
117            ),
118        )
119
120    kore_rhs0: Pattern = kast_to_kore(kast_defn, kompiled_kore, krule_rhs.kast, sort=top_level_k_sort)
121
122    kore_lhs = kompiled_kore.add_injections(kore_lhs0, sort=top_level_kore_sort)
123    kore_rhs = kompiled_kore.add_injections(kore_rhs0, sort=top_level_kore_sort)
124    prio = krule.priority
125    attrs = [App(symbol='priority', sorts=(), args=(String(str(prio)),))]
126    if 'label' in krule.att:
127        label = krule.att['label']
128        attrs.append(App(symbol='label', sorts=(), args=(String(label),)))
129    axiom = Axiom(
130        vars=(),
131        pattern=Rewrites(
132            sort=top_level_kore_sort,
133            left=kore_lhs,
134            right=kore_rhs,
135        ),
136        attrs=attrs,
137    )
138    return axiom

145def _kast_to_kore(term: KInner) -> Pattern:
146    stack: list = [term, []]
147    while True:
148        patterns = stack[-1]
149        term = stack[-2]
150        idx = len(patterns) - len(term.terms)
151        if not idx:
152            stack.pop()
153            stack.pop()
154            pattern = _kinner_to_kore(term, patterns)
155            if not stack:
156                return pattern
157            stack[-1].append(pattern)
158        else:
159            stack.append(term.terms[idx])
160            stack.append([])

163def _kinner_to_kore(kinner: KInner, patterns: list[Pattern]) -> Pattern:
164    match kinner:
165        case KToken():
166            assert not patterns
167            return _ktoken_to_kore(kinner)
168        case KVariable():
169            assert not patterns
170            return _kvariable_to_kore(kinner)
171        case KSequence():
172            return _ksequence_to_kore(kinner, patterns)
173        case KApply():
174            return _kapply_to_kore(kinner, patterns)
175        case _:
176            raise ValueError(f'Unsupported KInner: {kinner}')

179def _ktoken_to_kore(ktoken: KToken) -> DV:
180    value: String
181    if ktoken.sort == STRING:
182        value = String(pretty_string(ktoken))
183    elif ktoken.sort == BYTES:
184        value = String(pretty_bytes_str(ktoken))
185    else:
186        value = String(ktoken.token)
187
188    sort = _ksort_to_kore(ktoken.sort)
189
190    return DV(sort, value)

202def _ksequence_to_kore(kseq: KSequence, patterns: list[Pattern]) -> Pattern:
203    if not patterns:
204        return DOTK
205
206    unit: Pattern
207    args: list[Pattern]
208
209    last = patterns[-1]
210    if type(last) is EVar and last.sort == SORT_K:
211        unit = last
212        args = patterns[:-1]
213    else:
214        unit = DOTK
215        args = patterns
216
217    args.reverse()
218    return reduce(lambda x, y: App('kseq', (), (y, x)), args, unit)

228def _kapply_to_ml_quant(kapply: KApply, patterns: list[Pattern]) -> MLQuant:
229    label = kapply.label
230    symbol = ML_QUANT_LABELS[label.name]
231    sorts = tuple(_ksort_to_kore(ksort) for ksort in label.params)
232    return MLQuant.of(symbol, sorts, patterns)

235def _kapply_to_pattern(kapply: KApply, patterns: list[Pattern]) -> Pattern:
236    label = kapply.label
237    symbol = _label_to_kore(label.name)
238    sorts = tuple(_ksort_to_kore(ksort) for ksort in label.params)
239
240    if label.name in ML_PATTERN_LABELS:
241        return MLPattern.of(symbol, sorts, patterns)
242
243    return App(symbol, sorts, patterns)

263def _kore_to_kast(pattern: Pattern) -> KInner:
264    stack: list = [pattern.pattern if isinstance(pattern, Assoc) else pattern, []]
265    while True:
266        terms = stack[-1]
267        pattern = stack[-2]
268        patterns = pattern.patterns
269        idx = len(terms) - len(patterns)
270        if not idx:
271            stack.pop()
272            stack.pop()
273            term = _pattern_to_kast(pattern, terms)
274            if not stack:
275                return term
276            stack[-1].append(term)
277        else:
278            pattern = patterns[idx]
279            stack.append(pattern.pattern if isinstance(pattern, Assoc) else pattern)
280            stack.append([])

283def _pattern_to_kast(pattern: Pattern, terms: list[KInner]) -> KInner:
284    match pattern:
285        case DV(sort, String(value)):
286            assert not terms
287            if sort == KORE_STRING:
288                return stringToken(value)
289            if sort == KORE_BYTES:
290                return bytesToken_from_str(value)
291            return KToken(value, _sort_to_kast(sort))
292
293        case EVar(name, sort):
294            assert not terms
295            return KVariable(name=unmunge(name[3:]), sort=_sort_to_kast(sort))
296
297        case App(symbol, sorts, _):
298            if symbol == 'inj':
299                _, _ = sorts
300                (term,) = terms
301                return term
302
303            elif not sorts:
304                if symbol == 'dotk':
305                    () = terms
306                    return KSequence(())
307
308                elif symbol == 'kseq':
309                    _, _ = terms
310                    return KSequence(terms)
311
312                else:
313                    klabel = KLabel(unmunge(symbol[3:]))
314                    return KApply(klabel, terms)
315
316            # hardcoded polymorphic operators
317            elif (
318                symbol
319                == "Lbl'Hash'if'UndsHash'then'UndsHash'else'UndsHash'fi'Unds'K-EQUAL-SYNTAX'Unds'Sort'Unds'Bool'Unds'Sort'Unds'Sort"
320            ):
321                (sort,) = sorts
322                klabel = KLabel(unmunge(symbol[3:]), (_sort_to_kast(sort),))
323                return KApply(klabel, terms)
324
325            else:
326                raise ValueError(f'Unsupported polymorphic operator: {symbol}')
327
328        case Top(sort):
329            assert not terms
330            return mlTop(sort=_sort_to_kast(sort))
331
332        case Bottom(sort):
333            assert not terms
334            return mlBottom(sort=_sort_to_kast(sort))
335
336        case And(sort, _):
337            return mlAnd(terms, sort=_sort_to_kast(sort))
338
339        case Implies(sort, _, _):
340            larg, rarg = terms
341            return mlImplies(larg, rarg, sort=_sort_to_kast(sort))
342
343        case Not(sort, _):
344            (karg,) = terms
345            return mlNot(karg, sort=_sort_to_kast(sort))
346
347        case Exists(sort, EVar(vname, vsort), _):
348            kvar = KVariable(name=unmunge(vname[3:]), sort=_sort_to_kast(vsort))
349            (body,) = terms
350            return mlExists(kvar, body, sort=_sort_to_kast(sort))
351
352        case Equals(op_sort, sort, _, _):
353            larg, rarg = terms
354            return mlEquals(
355                larg,
356                rarg,
357                arg_sort=_sort_to_kast(op_sort),
358                sort=_sort_to_kast(sort),
359            )
360
361        case Ceil(op_sort, sort, _):
362            (karg,) = terms
363            return mlCeil(
364                karg,
365                arg_sort=_sort_to_kast(op_sort),
366                sort=_sort_to_kast(sort),
367            )
368
369        case _:
370            raise ValueError(f'Unsupported Pattern: {pattern}')

424def munge(label: str) -> str:
425    symbol = ''
426    quot = False
427    for c in label:
428        if c in MUNGE_TABLE:
429            symbol += "'" if not quot else ''
430            symbol += MUNGE_TABLE[c]
431            quot = True
432        else:
433            symbol += "'" if quot else ''
434            symbol += c
435            quot = False
436    symbol += "'" if quot else ''
437    return symbol

444def unmunged(it: Iterable[str]) -> Iterator[str]:
445    it = iter(it)
446
447    startquote = False
448    quote = False
449    endquote = False
450    buff: list[str] = []
451    cnt = 0  # len(buff)
452
453    for c in it:
454        if c == "'":
455            if startquote:
456                raise ValueError('Empty quoted section')
457            elif quote:
458                if cnt == 0:
459                    quote = False
460                    endquote = True
461                else:
462                    fragment = ''.join(buff)
463                    raise ValueError(f'Unexpected end of symbol: {fragment}')
464            elif endquote:
465                raise ValueError('Quoted sections next to each other')
466            else:
467                quote = True
468                startquote = True
469
470        else:
471            startquote = False
472            endquote = False
473            if quote:
474                if cnt == 3:
475                    buff.append(c)
476                    munged = ''.join(buff)
477                    buff = []
478                    cnt = 0
479                    if not munged in UNMUNGE_TABLE:
480                        raise ValueError(f'Unknown encoding "{munged}"')
481                    yield UNMUNGE_TABLE[munged]
482                else:
483                    buff += [c]
484                    cnt += 1
485            else:
486                yield c
487
488    if quote:
489        raise ValueError('Unterminated quoted section')
