 40def flatten_label(label: str, kast: KInner) -> list[KInner]:
 41    """Given a cons list, return a flat Python list of the elements.
 42
 43    -   Input: Cons operation to flatten.
 44    -   Output: Items of cons list.
 45    """
 46    if type(kast) is KApply and kast.label.name == label:
 47        items = (flatten_label(label, arg) for arg in kast.args)
 48        return [c for cs in items for c in cs]
 49    return [kast]

 52def is_term_like(kast: KInner) -> bool:
 53    non_term_found = False
 54
 55    def _is_term_like(_kast: KInner) -> None:
 56        nonlocal non_term_found
 57        match _kast:
 58            case KVariable(name, _):
 59                if name.startswith('@'):
 60                    non_term_found = True
 61            case KApply(KLabel(name, _), _):
 62                if name in {
 63                    '#Equals',
 64                    '#And',
 65                    '#Or',
 66                    '#Top',
 67                    '#Bottom',
 68                    '#Implies',
 69                    '#Not',
 70                    '#Ceil',
 71                    '#Forall',
 72                    '#Exists',
 73                }:
 74                    non_term_found = True
 75
 76    collect(_is_term_like, kast)
 77    return not non_term_found

 80def sort_assoc_label(label: str, kast: KInner) -> KInner:
 81    res: KInner | None = None
 82    if type(kast) is KApply and kast.label.name == label:
 83        terms = sorted(flatten_label(label, kast))
 84        for term in reversed(terms):
 85            if not res:
 86                res = term
 87            else:
 88                res = kast.label(term, res)
 89        assert res is not None
 90        return res
 91    return kast

116def ml_pred_to_bool(kast: KInner, unsafe: bool = False) -> KInner:
117    def _ml_constraint_to_bool(_kast: KInner) -> KInner:
118        if type(_kast) is KApply:
119            if _kast.label.name == '#Top':
120                return TRUE
121            if _kast.label.name == '#Bottom':
122                return FALSE
123            if _kast.label.name == '#Not' and len(_kast.args) == 1:
124                return notBool(_ml_constraint_to_bool(_kast.args[0]))
125            if _kast.label.name == '#And':
126                return andBool(map(_ml_constraint_to_bool, _kast.args))
127            if _kast.label.name == '#Or':
128                return orBool(map(_ml_constraint_to_bool, _kast.args))
129            if _kast.label.name == '#Implies' and len(_kast.args) == 2:
130                return impliesBool(_ml_constraint_to_bool(_kast.args[0]), _ml_constraint_to_bool(_kast.args[1]))
131            if _kast.label.name == '#Equals':
132                first, second = _kast.args
133                if first == TRUE:
134                    return second
135                if first == FALSE:
136                    return notBool(second)
137                if second == TRUE:
138                    return first
139                if second == FALSE:
140                    return notBool(first)
141                if type(first) in [KVariable, KToken]:
142                    return KApply('_==K_', _kast.args)
143                if type(first) is KSequence and type(second) is KSequence:
144                    if first.arity == 1 and second.arity == 1:
145                        return KApply('_==K_', (first.items[0], second.items[0]))
146                if is_term_like(first) and is_term_like(second):
147                    return KApply('_==K_', first, second)
148            if unsafe:
149                if _kast.label.name == '#Equals':
150                    return KApply('_==K_', _kast.args)
151                if _kast.label.name == '#Ceil':
152                    ceil_var = abstract_term_safely(_kast, base_name='Ceil')
153                    _LOGGER.warning(f'Converting #Ceil condition to variable {ceil_var.name}: {_kast}')
154                    return ceil_var
155                if _kast.label.name == '#Exists':
156                    exists_var = abstract_term_safely(_kast, base_name='Exists')
157                    _LOGGER.warning(f'Converting #Exists condition to variable {exists_var.name}: {_kast}')
158                    return exists_var
159        raise ValueError(f'Could not convert ML predicate to sort Bool: {_kast}')
160
161    return _ml_constraint_to_bool(kast)

164def simplify_bool(k: KInner) -> KInner:
165    if k is None:
166        return None
167
168    # fmt: off
169    simplify_rules = [ (KApply('_==K_', [KVariable('#LHS'), TRUE]), KVariable('#LHS'))
170                     , (KApply('_==K_', [TRUE, KVariable('#RHS')]), KVariable('#RHS'))
171                     , (KApply('_==K_', [KVariable('#LHS'), FALSE]), notBool(KVariable('#LHS')))
172                     , (KApply('_==K_', [FALSE, KVariable('#RHS')]), notBool(KVariable('#RHS')))
173                     , (notBool(FALSE), TRUE)
174                     , (notBool(TRUE), FALSE)
175                     , (notBool(KApply('_==K_'    , [KVariable('#V1'), KVariable('#V2')])), KApply('_=/=K_'   , [KVariable('#V1'), KVariable('#V2')]))
176                     , (notBool(KApply('_=/=K_'   , [KVariable('#V1'), KVariable('#V2')])), KApply('_==K_'    , [KVariable('#V1'), KVariable('#V2')]))
177                     , (notBool(KApply('_==Int_'  , [KVariable('#V1'), KVariable('#V2')])), KApply('_=/=Int_' , [KVariable('#V1'), KVariable('#V2')]))
178                     , (notBool(KApply('_=/=Int_' , [KVariable('#V1'), KVariable('#V2')])), KApply('_==Int_'  , [KVariable('#V1'), KVariable('#V2')]))
179                     , (andBool([TRUE, KVariable('#REST')]), KVariable('#REST'))
180                     , (andBool([KVariable('#REST'), TRUE]), KVariable('#REST'))
181                     , (andBool([FALSE, KVariable('#REST')]), FALSE)
182                     , (andBool([KVariable('#REST'), FALSE]), FALSE)
183                     , (orBool([FALSE, KVariable('#REST')]), KVariable('#REST'))
184                     , (orBool([KVariable('#REST'), FALSE]), KVariable('#REST'))
185                     , (orBool([TRUE, KVariable('#REST')]), TRUE)
186                     , (orBool([KVariable('#REST'), TRUE]), TRUE)
187                     ]
188    # fmt: on
189
190    new_k = k
191    for rule in simplify_rules:
192        rewrite = KRewrite(*rule)
193        new_k = rewrite(new_k)
194    return new_k

205def extract_subst(term: KInner) -> tuple[Subst, KInner]:
206    def _subst_for_terms(term1: KInner, term2: KInner) -> Subst | None:
207        if type(term1) is KVariable and type(term2) not in {KToken, KVariable}:
208            return Subst({term1.name: term2})
209        if type(term2) is KVariable and type(term1) not in {KToken, KVariable}:
210            return Subst({term2.name: term1})
211        return None
212
213    def _extract_subst(conjunct: KInner) -> Subst | None:
214        if type(conjunct) is KApply:
215            if conjunct.label.name == '#Equals':
216                subst = _subst_for_terms(conjunct.args[0], conjunct.args[1])
217
218                if subst is not None:
219                    return subst
220
221                if (
222                    conjunct.args[0] == TRUE
223                    and type(conjunct.args[1]) is KApply
224                    and conjunct.args[1].label.name in {'_==K_', '_==Int_'}
225                ):
226                    subst = _subst_for_terms(conjunct.args[1].args[0], conjunct.args[1].args[1])
227
228                    if subst is not None:
229                        return subst
230
231        return None
232
233    conjuncts = flatten_label('#And', term)
234    subst = Subst()
235    rem_conjuncts: list[KInner] = []
236
237    for conjunct in conjuncts:
238        new_subst = _extract_subst(conjunct)
239        if new_subst is None:
240            rem_conjuncts.append(conjunct)
241        else:
242            new_subst = subst.union(new_subst)
243            if new_subst is None:
244                raise ValueError('Conflicting substitutions')  # TODO handle this case
245            subst = new_subst
246
247    return subst, mlAnd(rem_conjuncts)

262def propagate_up_constraints(k: KInner) -> KInner:
263    def _propagate_up_constraints(_k: KInner) -> KInner:
264        if not (type(_k) is KApply and _k.label.name == '#Or'):
265            return _k
266        top_sort = _k.label.params[0]
267        conjuncts1 = flatten_label('#And', _k.args[0])
268        conjuncts2 = flatten_label('#And', _k.args[1])
269        (common1, l1, r1) = find_common_items(conjuncts1, conjuncts2)
270        (common2, r2, l2) = find_common_items(r1, l1)
271        common = common1 + common2
272        if len(common) == 0:
273            return _k
274        conjunct1 = mlAnd(l2, sort=top_sort)
275        conjunct2 = mlAnd(r2, sort=top_sort)
276        disjunct = mlOr([conjunct1, conjunct2], sort=top_sort)
277        return mlAnd([disjunct] + common, sort=top_sort)
278
279    return bottom_up(_propagate_up_constraints, k)

282def split_config_and_constraints(kast: KInner) -> tuple[KInner, KInner]:
283    conjuncts = flatten_label('#And', kast)
284    term = None
285    constraints = []
286    for c in conjuncts:
287        if type(c) is KApply and c.is_cell:
288            if term:
289                raise ValueError(f'Found two configurations in pattern:\n\n{term}\n\nand\n\n{c}')
290            term = c
291        else:
292            constraints.append(c)
293    if not term:
294        raise ValueError(f'Could not find configuration for: {kast}')
295    return (term, mlAnd(constraints, GENERATED_TOP_CELL))

303def split_config_from(configuration: KInner) -> tuple[KInner, dict[str, KInner]]:
304    """Split the substitution from a given configuration.
305
306    Given an input configuration `config`, will return a tuple `(symbolic_config, subst)`, where:
307
308        1.  `config == substitute(symbolic_config, subst)`
309        2.  `symbolic_config` is the same configuration structure, but where the contents of leaf cells is replaced with a fresh KVariable.
310        3.  `subst` is the substitution for the generated KVariables back to the original configuration contents.
311    """
312    initial_substitution = {}
313
314    def _replace_with_var(k: KInner) -> KInner:
315        if type(k) is KApply and k.is_cell:
316            if k.arity == 1 and not (type(k.args[0]) is KApply and k.args[0].is_cell):
317                config_var = cell_label_to_var_name(k.label.name)
318                initial_substitution[config_var] = k.args[0]
319                return KApply(k.label, [KVariable(config_var)])
320        return k
321
322    symbolic_config = top_down(_replace_with_var, configuration)
323    return (symbolic_config, initial_substitution)

326def collapse_dots(kast: KInner) -> KInner:
327    """Given a configuration with structural frames `...`, minimize the structural frames needed.
328
329    -   Input: a configuration, potentially with structural frames.
330    -   Output: the same configuration, with the amount of structural framing minimized.
331    """
332
333    def _collapse_dots(_kast: KInner) -> KInner:
334        if type(_kast) is KApply:
335            if _kast.is_cell and _kast.arity == 1 and _kast.args[0] == DOTS:
336                return DOTS
337            new_args = [arg for arg in _kast.args if arg != DOTS]
338            if _kast.is_cell and len(new_args) == 0:
339                return DOTS
340            if len(new_args) < len(_kast.args):
341                new_args.append(DOTS)
342            return _kast.let(args=new_args)
343        elif type(_kast) is KRewrite:
344            if _kast.lhs == DOTS:
345                return DOTS
346        return _kast
347
348    return bottom_up(_collapse_dots, kast)

351def push_down_rewrites(kast: KInner) -> KInner:
352    def _push_down_rewrites(_kast: KInner) -> KInner:
353        if type(_kast) is KRewrite:
354            lhs = _kast.lhs
355            rhs = _kast.rhs
356            if lhs == rhs:
357                return lhs
358            if type(lhs) is KVariable and type(rhs) is KVariable and lhs.name == rhs.name:
359                return lhs
360            if type(lhs) is KApply and type(rhs) is KApply and lhs.label == rhs.label and lhs.arity == rhs.arity:
361                new_args = [
362                    KRewrite(left_arg, right_arg) for left_arg, right_arg in zip(lhs.args, rhs.args, strict=True)
363                ]
364                return lhs.let(args=new_args)
365            if type(lhs) is KSequence and type(rhs) is KSequence and lhs.arity > 0 and rhs.arity > 0:
366                if lhs.arity == 1 and rhs.arity == 1:
367                    return KRewrite(lhs.items[0], rhs.items[0])
368                if lhs.items[0] == rhs.items[0]:
369                    lower_rewrite = _push_down_rewrites(KRewrite(KSequence(lhs.items[1:]), KSequence(rhs.items[1:])))
370                    return KSequence([lhs.items[0], lower_rewrite])
371                if lhs.items[-1] == rhs.items[-1]:
372                    lower_rewrite = _push_down_rewrites(
373                        KRewrite(KSequence(lhs.items[0:-1]), KSequence(rhs.items[0:-1]))
374                    )
375                    return KSequence([lower_rewrite, lhs.items[-1]])
376            if (
377                type(lhs) is KSequence
378                and lhs.arity > 0
379                and type(lhs.items[-1]) is KVariable
380                and type(rhs) is KVariable
381                and lhs.items[-1] == rhs
382            ):
383                return KSequence([KRewrite(KSequence(lhs.items[0:-1]), KSequence([])), rhs])
384        return _kast
385
386    return top_down(_push_down_rewrites, kast)

421def useless_vars_to_dots(kast: KInner, keep_vars: Iterable[str] = ()) -> KInner:
422    """Structurally abstract away useless variables.
423
424    -   Input: kast term, and a requires clause and ensures clause.
425    -   Output: kast term with the useless vars structurally abstracted.
426    """
427    num_occs = count_vars(kast) + Counter(keep_vars)
428
429    def _collapse_useless_vars(_kast: KInner) -> KInner:
430        if type(_kast) is KApply and _kast.is_cell:
431            new_args: list[KInner] = []
432            for arg in _kast.args:
433                if type(arg) is KVariable and num_occs[arg.name] == 1:
434                    new_args.append(DOTS)
435                else:
436                    new_args.append(arg)
437            return _kast.let(args=new_args)
438        return _kast
439
440    return bottom_up(_collapse_useless_vars, kast)

488def minimize_term(
489    term: KInner, keep_vars: Iterable[str] = (), abstract_labels: Collection[str] = (), keep_cells: Collection[str] = ()
490) -> KInner:
491    """Minimize a K term for pretty-printing.
492
493    -   Input: kast term, and optionally requires and ensures clauses with constraints.
494    -   Output: kast term minimized.
495        -   Variables only used once will be removed.
496        -   Unused cells will be abstracted.
497        -   Attempt to remove useless conditions.
498    """
499
500    term = inline_cell_maps(term)
501    term = remove_semantic_casts(term)
502    term = useless_vars_to_dots(term, keep_vars=keep_vars)
503
504    if keep_cells:
505        term = extract_cells(term, keep_cells)
506    else:
507        term = labels_to_dots(term, abstract_labels)
508
509    term = collapse_dots(term)
510
511    return term

514def minimize_rule(rule: RL, keep_vars: Iterable[str] = ()) -> RL:
515    """Minimize a K rule or claim for pretty-printing.
516
517    -   Input: kast representing a K rule or claim.
518    -   Output: kast with the rule or claim minimized:
519        -   Variables only used once will be removed.
520        -   Unused cells will be abstracted.
521        -   Attempt to remove useless side-conditions.
522    """
523    body = rule.body
524    requires = rule.requires
525    ensures = rule.ensures
526
527    requires = andBool(flatten_label('_andBool_', requires))
528    requires = simplify_bool(requires)
529
530    ensures = andBool(flatten_label('_andBool_', ensures))
531    ensures = simplify_bool(ensures)
532
533    constrained_vars = list(keep_vars)
534    constrained_vars = constrained_vars + free_vars(requires)
535    constrained_vars = constrained_vars + free_vars(ensures)
536    body = minimize_term(body, keep_vars=constrained_vars)
537
538    return rule.let(body=body, requires=requires, ensures=ensures)

595def apply_existential_substitutions(state: KInner, constraints: Iterable[KInner]) -> tuple[KInner, Iterable[KInner]]:
596    pattern = mlEqualsTrue(KApply('_==K_', [KVariable('#VAR'), KVariable('#VAL')]))
597    subst = {}
598    new_constraints = []
599    for c in constraints:
600        match = pattern.match(c)
601        if match is not None and type(match['#VAR']) is KVariable and match['#VAR'].name.startswith('?'):
602            subst[match['#VAR'].name] = match['#VAL']
603        else:
604            new_constraints.append(c)
605    return (Subst(subst)(state), [Subst(subst)(c) for c in new_constraints])

608def indexed_rewrite(kast: KInner, rewrites: Iterable[KRewrite]) -> KInner:
609    token_rewrites: list[KRewrite] = []
610    apply_rewrites: dict[str, list[KRewrite]] = {}
611    other_rewrites: list[KRewrite] = []
612    for r in rewrites:
613        if type(r.lhs) is KToken:
614            token_rewrites.append(r)
615        elif type(r.lhs) is KApply:
616            if r.lhs.label.name in token_rewrites:
617                apply_rewrites[r.lhs.label.name].append(r)
618            else:
619                apply_rewrites[r.lhs.label.name] = [r]
620        else:
621            other_rewrites.append(r)
622
623    def _apply_rewrites(_kast: KInner) -> KInner:
624        if type(_kast) is KToken:
625            for tr in token_rewrites:
626                _kast = tr.apply_top(_kast)
627        elif type(_kast) is KApply:
628            if _kast.label.name in apply_rewrites:
629                for ar in apply_rewrites[_kast.label.name]:
630                    _kast = ar.apply_top(_kast)
631        else:
632            for _or in other_rewrites:
633                _kast = _or.apply_top(_kast)
634        return _kast
635
636    orig_kast: KInner = kast
637    new_kast: KInner | None = None
638    while orig_kast != new_kast:
639        if new_kast is None:
640            new_kast = orig_kast
641        else:
642            orig_kast = new_kast
643        new_kast = bottom_up(_apply_rewrites, new_kast)
644    return new_kast

647def undo_aliases(definition: KDefinition, kast: KInner) -> KInner:
648    aliases = []
649    for rule in definition.alias_rules:
650        rewrite = rule.body
651        if type(rewrite) is not KRewrite:
652            raise ValueError(f'Expected KRewrite as alias body, found: {rewrite}')
653        if rule.requires is not None and rule.requires != TRUE:
654            raise ValueError(f'Expended empty requires clause on alias, found: {rule.requires}')
655        if rule.ensures is not None and rule.ensures != TRUE:
656            raise ValueError(f'Expended empty ensures clause on alias, found: {rule.ensures}')
657        aliases.append(KRewrite(rewrite.rhs, rewrite.lhs))
658    return indexed_rewrite(kast, aliases)

661def rename_generated_vars(term: KInner) -> KInner:
662    vars: set[str] = set(free_vars(term))
663    cell_stack: list[str] = []
664
665    def _rename_vars(k: KInner) -> KInner:
666        if type(k) is KApply and k.is_cell:
667            cell_stack.append(cell_label_to_var_name(k.label.name))
668            res = k.map_inner(_rename_vars)
669            cell_stack.pop()
670            return res
671
672        if type(k) is KVariable and k.name.startswith(('_Gen', '?_Gen', '_DotVar', '?_DotVar')):
673            if not cell_stack:
674                return k
675            cell_name = cell_stack[-1]
676            new_var = abstract_term_safely(k, base_name=cell_name, sort=k.sort, existing_var_names=vars)
677            vars.add(new_var.name)
678            return new_var
679
680        return k.map_inner(_rename_vars)
681
682    return _rename_vars(term)

685def remove_useless_constraints(constraints: Iterable[KInner], initial_vars: Iterable[str]) -> list[KInner]:
686    """Given a list of constraints and a list of variables, return an updated list with only constraints that depend on these variables (directly or indirectly).
687
688    :param constraints: Original list of constraints to remove from.
689    :param initial_vars: Initial list of variables to keep constraints for.
690    :return: A list of constraints with only those constraints that contain the initial variables or variables that depend on those through other constraints in the list.
691    """
692    used_vars = list(initial_vars)
693    prev_len_used_vars = 0
694    new_constraints = []
695    while len(used_vars) > prev_len_used_vars:
696        prev_len_used_vars = len(used_vars)
697        for c in constraints:
698            if c not in new_constraints:
699                new_vars = free_vars(c)
700                if any(v in used_vars for v in new_vars):
701                    new_constraints.append(c)
702                    used_vars.extend(new_vars)
703        used_vars = list(set(used_vars))
704    return new_constraints

734def build_rule(
735    rule_id: str,
736    init_config: KInner,
737    final_config: KInner,
738    init_constraints: Iterable[KInner] = (),
739    final_constraints: Iterable[KInner] = (),
740    priority: int | None = None,
741    keep_vars: Iterable[str] = (),
742) -> tuple[KRule, Subst]:
743    """Return a `KRule` between the supplied initial and final states.
744
745    :param rule_id: Label to give the rule.
746    :param init_config: State to put on LHS of the rule.
747    :param final_config: State to put on RHS of the rule.
748    :param init_constraints: Constraints to use as `requires` clause.
749    :param final_constraints: Constraints to use as `ensures` clause.
750    :param priority: Rule priority to give to the generated `KRule`.
751    :param keep_vars: Variables to leave in the side-conditions even if not bound in the configuration.
752    :return: tuple `claim: KRule, var_map: Subst` such that:
753      - `rule`: A `KRule` with variable naming conventions applied so that it should be parseable by K frontend.
754      - `var_map`: The variable renamings that happened to make the claim parseable by K frontend (which can be undone to recover original variables).
755    """
756    init_constraints = list(init_constraints)
757    final_constraints = [c for c in final_constraints if c not in init_constraints]
758    init_term = mlAnd([init_config] + init_constraints)
759    final_term = mlAnd([final_config] + final_constraints)
760
761    lhs_vars = free_vars(init_term)
762    rhs_vars = free_vars(final_term)
763    var_occurrences = count_vars(
764        mlAnd(
765            [push_down_rewrites(KRewrite(init_config, final_config))] + init_constraints + final_constraints,
766            GENERATED_TOP_CELL,
767        )
768    )
769    v_subst: dict[str, KVariable] = {}
770    vremap_subst: dict[str, KVariable] = {}
771    for v in var_occurrences:
772        new_v = v
773        if var_occurrences[v] == 1:
774            new_v = '_' + new_v
775        if v in rhs_vars and v not in lhs_vars:
776            new_v = '?' + new_v
777        if new_v != v:
778            v_subst[v] = KVariable(new_v)
779            vremap_subst[new_v] = KVariable(v)
780
781    new_init_config = Subst(v_subst)(init_config)
782    new_init_constraints = [Subst(v_subst)(c) for c in init_constraints]
783    new_final_config, new_final_constraints = apply_existential_substitutions(
784        Subst(v_subst)(final_config), [Subst(v_subst)(c) for c in final_constraints]
785    )
786
787    rule_body = push_down_rewrites(KRewrite(new_init_config, new_final_config))
788    rule_requires = simplify_bool(ml_pred_to_bool(mlAnd(new_init_constraints)))
789    rule_ensures = simplify_bool(ml_pred_to_bool(mlAnd(new_final_constraints)))
790    att_entries = [] if priority is None else [Atts.PRIORITY(str(priority))]
791    rule_att = KAtt(entries=att_entries)
792
793    rule = KRule(rule_body, requires=rule_requires, ensures=rule_ensures, att=rule_att)
794    rule = rule.update_atts([Atts.LABEL(rule_id)])
795
796    return (rule, Subst(vremap_subst))
