 40    def is_explorable(self, node_id: NodeIdLike) -> bool:
 41        return (
 42            self.kcfg.is_leaf(node_id)
 43            and not self.is_terminal(node_id)
 44            and not self.kcfg.is_stuck(node_id)
 45            and not self.kcfg.is_vacuous(node_id)
 46        )
