195    def _info_text(self) -> str:
196        term_str = '✅' if self._term_on else '❌'
197        constraint_str = '✅' if self._constraint_on else '❌'
198        custom_str = '' if self._custom_view is None else f'{"✅" if self._custom_on else "❌"} Custom View.'
199        minimize_str = '✅' if self._minimize else '❌'
200        element_str = 'NOTHING'
201        if type(self._element) is KCFG.Node:
202            element_str = f'node({shorten_hashes(self._element.id)})'
203        elif type(self._element) is KCFG.Edge:
204            element_str = f'edge({shorten_hashes(self._element.source.id)},{shorten_hashes(self._element.target.id)})'
205        elif type(self._element) is KCFG.Cover:
206            element_str = f'cover({shorten_hashes(self._element.source.id)},{shorten_hashes(self._element.target.id)})'
207        return f'{element_str} selected. {minimize_str} Minimize Output. {term_str} Term View. {constraint_str} Constraint View. {custom_str}'

216    def toggle_option(self, field: str) -> bool:
217        assert field in ['minimize', 'term_on', 'constraint_on', 'custom_on']
218        field_attr = f'_{field}'
219        old_value = getattr(self, field_attr)
220        new_value = not old_value
221        # Do not turn on custom view if it's not available
222        if field == 'custom_on' and self._custom_view is None:
223            new_value = False
224        setattr(self, field_attr, new_value)
225        self._update()
226        return new_value

242    def _update(self) -> None:
243        def _boolify(c: KInner) -> KInner:
244            if type(c) is KApply and c.label.name == '#Equals' and c.args[0] == TRUE:
245                return c.args[1]
246            else:
247                return c
248
249        def _cterm_text(cterm: CTerm) -> tuple[str, str]:
250            config = cterm.config
251            constraints = map(_boolify, cterm.constraints)
252            if self._minimize:
253                config = minimize_term(config)
254            return (self._kprint.pretty_print(config), '\n'.join(self._kprint.pretty_print(c) for c in constraints))
255
256        term_str = 'Term'
257        constraint_str = 'Constraint'
258        custom_str = 'Custom'
259
260        if self._element is not None:
261            if type(self._element) is KCFG.Node:
262                term_str, constraint_str = _cterm_text(self._element.cterm)
263
264            elif type(self._element) is KCFG.Edge:
265                config_source, *constraints_source = self._element.source.cterm
266                config_target, *constraints_target = self._element.target.cterm
267                constraints_new = [c for c in constraints_target if c not in constraints_source]
268                config = push_down_rewrites(KRewrite(config_source, config_target))
269                crewrite = CTerm(config, constraints_new)
270                term_str, constraint_str = _cterm_text(crewrite)
271
272            elif type(self._element) is KCFG.Cover:
273                subst_equalities = map(_boolify, flatten_label('#And', self._element.csubst.subst.ml_pred))
274                constraints = map(_boolify, flatten_label('#And', self._element.csubst.constraint))
275                term_str = '\n'.join(self._kprint.pretty_print(se) for se in subst_equalities)
276                constraint_str = '\n'.join(self._kprint.pretty_print(c) for c in constraints)
277
278            elif type(self._element) is KCFG.Split:
279                term_strs = [f'split: {shorten_hashes(self._element.source.id)}']
280                for target_id, csubst in self._element.splits.items():
281                    term_strs.append('')
282                    term_strs.append(f'  - {shorten_hashes(target_id)}')
283                    if len(csubst.subst) > 0:
284                        subst_equalities = map(_boolify, flatten_label('#And', csubst.subst.ml_pred))
285                        term_strs.extend(f'    {self._kprint.pretty_print(cline)}' for cline in subst_equalities)
286                    if len(csubst.constraints) > 0:
287                        constraints = map(_boolify, flatten_label('#And', csubst.constraint))
288                        term_strs.extend(f'    {self._kprint.pretty_print(cline)}' for cline in constraints)
289                term_str = '\n'.join(term_strs)
290
291            elif type(self._element) is KCFG.NDBranch:
292                term_strs = [f'ndbranch: {shorten_hashes(self._element.source.id)}']
293                for target in self._element.targets:
294                    term_strs.append('')
295                    term_strs.append(f'  - {shorten_hashes(target.id)}')
296                    term_strs.append('    (1 step)')
297                term_str = '\n'.join(term_strs)
298
299            if self._custom_view is not None:
300                # To appease the type-checker
301                if type(self._element) is KCFG.Node:
302                    custom_str = '\n'.join(self._custom_view(self._element))
303                elif isinstance(self._element, KCFG.Successor):
304                    custom_str = '\n'.join(self._custom_view(self._element))
305
306        self.query_one('#info', Info).text = self._info_text()
307        self.query_one('#term', Term).text = term_str
308        self.query_one('#constraint', Constraint).text = constraint_str
309        if self._custom_view is not None:
310            self.query_one('#custom', Custom).text = custom_str

368    def on_graph_chunk_selected(self, message: GraphChunk.Selected) -> None:
369        self.query_one('#behavior').focus()
370
371        if message.chunk_id.startswith('node_'):
372            self._selected_chunk = message.chunk_id
373            node, *_ = message.chunk_id[5:].split('_')
374            node_id = int(node)
375            self.query_one('#node-view', NodeView).update(self._kcfg.node(node_id))
376
377        elif message.chunk_id.startswith('edge_'):
378            self._selected_chunk = None
379            node_source, node_target, *_ = message.chunk_id[5:].split('_')
380            source_id = int(node_source)
381            target_id = int(node_target)
382            edge = single(self._kcfg.edges(source_id=source_id, target_id=target_id))
383            self.query_one('#node-view', NodeView).update(edge)
384
385        elif message.chunk_id.startswith('cover_'):
386            self._selected_chunk = None
387            node_source, node_target, *_ = message.chunk_id[6:].split('_')
388            source_id = int(node_source)
389            target_id = int(node_target)
390            cover = single(self._kcfg.covers(source_id=source_id, target_id=target_id))
391            self.query_one('#node-view', NodeView).update(cover)
392
393        elif message.chunk_id.startswith('split_'):
394            self._selected_chunk = None
395            node_source, node_target, *_ = message.chunk_id[6:].split('_')
396            source_id = int(node_source)

418    def action_keystroke(self, key: str) -> None:
419        if key == 'h':
420            if self._selected_chunk is not None and self._selected_chunk.startswith('node_'):
421                node_id = self._selected_chunk[5:]
422                self._hidden_chunks.append(self._selected_chunk)
423                self.query_one(f'#{self._selected_chunk}', GraphChunk).add_class('hidden')
424                self.query_one('#info', Info).text = f'HIDDEN: node({shorten_hashes(node_id)})'
425        elif key == 'H':
426            for hc in self._hidden_chunks:
427                self.query_one(f'#{hc}', GraphChunk).remove_class('hidden')
428            node_ids = [nid[5:] for nid in self._hidden_chunks]
429            self.query_one('#info', Info).text = f'UNHIDDEN: nodes({shorten_hashes(node_ids)})'
430            self._hidden_chunks = []
431        elif key in ['term', 'constraint']:
432            self.query_one('#node-view', NodeView).toggle_view(key)
433        elif key == 'custom' and self._custom_view is not None:
434            self.query_one('#node-view', NodeView).toggle_view(key)
435        elif key in ['minimize']:
436            self.query_one('#node-view', NodeView).toggle_option(key)
