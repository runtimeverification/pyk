 77    def cterm_execute(
 78        self,
 79        cterm: CTerm,
 80        depth: int | None = None,
 81        cut_point_rules: Iterable[str] | None = None,
 82        terminal_rules: Iterable[str] | None = None,
 83        module_name: str | None = None,
 84    ) -> CTermExecute:
 85        _LOGGER.debug(f'Executing: {cterm}')
 86        kore = self.kprint.kast_to_kore(cterm.kast, GENERATED_TOP_CELL)
 87        response = self._kore_client.execute(
 88            kore,
 89            max_depth=depth,
 90            cut_point_rules=cut_point_rules,
 91            terminal_rules=terminal_rules,
 92            module_name=module_name,
 93            log_successful_rewrites=True,
 94            log_failed_rewrites=self._trace_rewrites,
 95            log_successful_simplifications=self._trace_rewrites,
 96            log_failed_simplifications=self._trace_rewrites,
 97        )
 98
 99        if isinstance(response, AbortedResult):
100            unknown_predicate = response.unknown_predicate.text if response.unknown_predicate else None
101            raise ValueError(f'Backend responded with aborted state. Unknown predicate: {unknown_predicate}')
102
103        state = CTerm.from_kast(self.kprint.kore_to_kast(response.state.kore))
104        resp_next_states = response.next_states or ()
105        next_states = tuple(CTerm.from_kast(self.kprint.kore_to_kast(ns.kore)) for ns in resp_next_states)
106
107        assert all(not cterm.is_bottom for cterm in next_states)
108        assert len(next_states) != 1 or response.reason is StopReason.CUT_POINT_RULE
109
110        return CTermExecute(
111            state=state,
112            next_states=next_states,
113            depth=response.depth,
114            vacuous=response.reason is StopReason.VACUOUS,
115            logs=response.logs,
116        )

132    def cterm_get_model(self, cterm: CTerm, module_name: str | None = None) -> Subst | None:
133        _LOGGER.info(f'Getting model: {cterm}')
134        kore = self.kprint.kast_to_kore(cterm.kast, GENERATED_TOP_CELL)
135        result = self._kore_client.get_model(kore, module_name=module_name)
136        if type(result) is UnknownResult:
137            _LOGGER.debug('Result is Unknown')
138            return None
139        elif type(result) is UnsatResult:
140            _LOGGER.debug('Result is UNSAT')
141            return None
142        elif type(result) is SatResult:
143            _LOGGER.debug('Result is SAT')
144            if not result.model:
145                return Subst({})
146            model_subst = self.kprint.kore_to_kast(result.model)
147            try:
148                return Subst.from_pred(model_subst)
149            except ValueError as err:
150                raise AssertionError(f'Received a non-substitution from get-model endpoint: {model_subst}') from err
151
152        else:
153            raise AssertionError('Received an invalid response from get-model endpoint')

155    def cterm_implies(
156        self,
157        antecedent: CTerm,
158        consequent: CTerm,
159        bind_universally: bool = False,
160    ) -> CSubst | None:
161        _LOGGER.debug(f'Checking implication: {antecedent} #Implies {consequent}')
162        _consequent = consequent.kast
163        fv_antecedent = free_vars(antecedent.kast)
164        unbound_consequent = [v for v in free_vars(_consequent) if v not in fv_antecedent]
165        if len(unbound_consequent) > 0:
166            bind_text, bind_label = ('existentially', '#Exists')
167            if bind_universally:
168                bind_text, bind_label = ('universally', '#Forall')
169            _LOGGER.debug(f'Binding variables in consequent {bind_text}: {unbound_consequent}')
170            for uc in unbound_consequent:
171                _consequent = KApply(KLabel(bind_label, [GENERATED_TOP_CELL]), [KVariable(uc), _consequent])
172        antecedent_kore = self.kprint.kast_to_kore(antecedent.kast, GENERATED_TOP_CELL)
173        consequent_kore = self.kprint.kast_to_kore(_consequent, GENERATED_TOP_CELL)
174        result = self._kore_client.implies(antecedent_kore, consequent_kore)
175        if not result.satisfiable:
176            if result.substitution is not None:
177                _LOGGER.debug(f'Received a non-empty substitution for unsatisfiable implication: {result.substitution}')
178            if result.predicate is not None:
179                _LOGGER.debug(f'Received a non-empty predicate for unsatisfiable implication: {result.predicate}')
180            return None
181        if result.substitution is None:
182            raise ValueError('Received empty substutition for satisfiable implication.')
183        if result.predicate is None:
184            raise ValueError('Received empty predicate for satisfiable implication.')
185        ml_subst = self.kprint.kore_to_kast(result.substitution)
186        ml_pred = self.kprint.kore_to_kast(result.predicate) if result.predicate is not None else mlTop()
187        ml_preds = flatten_label('#And', ml_pred)
188        if is_top(ml_subst):
189            return CSubst(subst=Subst({}), constraints=ml_preds)
190        subst_pattern = mlEquals(KVariable('###VAR'), KVariable('###TERM'))
191        _subst: dict[str, KInner] = {}
192        for subst_pred in flatten_label('#And', ml_subst):
193            m = subst_pattern.match(subst_pred)
194            if m is not None and type(m['###VAR']) is KVariable:
195                _subst[m['###VAR'].name] = m['###TERM']
196            else:
197                raise AssertionError(f'Received a non-substitution from implies endpoint: {subst_pred}')
198        return CSubst(subst=Subst(_subst), constraints=ml_preds)

200    def implication_failure_reason(self, antecedent: CTerm, consequent: CTerm) -> tuple[bool, str]:
201        def no_cell_rewrite_to_dots(term: KInner) -> KInner:
202            def _no_cell_rewrite_to_dots(_term: KInner) -> KInner:
203                if type(_term) is KApply and _term.is_cell:
204                    lhs = extract_lhs(_term)
205                    rhs = extract_rhs(_term)
206                    if lhs == rhs:
207                        return KApply(_term.label, [abstract_term_safely(lhs, base_name=_term.label.name)])
208                return _term
209
210            return bottom_up(_no_cell_rewrite_to_dots, term)
211
212        def _is_cell_subst(csubst: KInner) -> bool:
213            if type(csubst) is KApply and csubst.label.name == '_==K_':
214                csubst_arg = csubst.args[0]
215                if type(csubst_arg) is KVariable and csubst_arg.name.endswith('_CELL'):
216                    return True
217            return False
218
219        def _is_negative_cell_subst(constraint: KInner) -> bool:
220            constraint_bool = ml_pred_to_bool(constraint)
221            if type(constraint_bool) is KApply and constraint_bool.label.name == 'notBool_':
222                negative_constraint = constraint_bool.args[0]
223                if type(negative_constraint) is KApply and negative_constraint.label.name == '_andBool_':
224                    constraints = flatten_label('_andBool_', negative_constraint)
225                    cell_constraints = list(filter(_is_cell_subst, constraints))
226                    if len(cell_constraints) > 0:
227                        return True
228            return False
229
230        def replace_rewrites_with_implies(kast: KInner) -> KInner:
231            def _replace_rewrites_with_implies(_kast: KInner) -> KInner:
232                if type(_kast) is KRewrite:
233                    return mlImplies(_kast.lhs, _kast.rhs)
234                return _kast
235
236            return bottom_up(_replace_rewrites_with_implies, kast)
237
238        config_match = self.cterm_implies(CTerm.from_kast(antecedent.config), CTerm.from_kast(consequent.config))
239        if config_match is None:
240            failing_cells = []
241            curr_cell_match = Subst({})
242            for cell in antecedent.cells:
243                antecedent_cell = antecedent.cell(cell)
244                consequent_cell = consequent.cell(cell)
245                cell_match = consequent_cell.match(antecedent_cell)
246                if cell_match is not None:
247                    _curr_cell_match = curr_cell_match.union(cell_match)
248                    if _curr_cell_match is not None:
249                        curr_cell_match = _curr_cell_match
250                        continue
251                failing_cell = push_down_rewrites(KRewrite(antecedent_cell, consequent_cell))
252                failing_cell = no_cell_rewrite_to_dots(failing_cell)
253                failing_cell = replace_rewrites_with_implies(failing_cell)
254                failing_cells.append((cell, failing_cell))
255            failing_cells_str = '\n'.join(
256                f'{cell}: {self.kprint.pretty_print(minimize_term(rew))}' for cell, rew in failing_cells
257            )
258            return (
259                False,
260                f'Structural matching failed, the following cells failed individually (antecedent #Implies consequent):\n{failing_cells_str}',
261            )
262        else:
263            consequent_constraints = list(
264                filter(lambda x: not CTerm._is_spurious_constraint(x), map(config_match.subst, consequent.constraints))
265            )
266            impl = CTerm._ml_impl(antecedent.constraints, consequent_constraints)
267            if impl != mlTop(k.GENERATED_TOP_CELL):
268                fail_str = self.kprint.pretty_print(impl)
269                negative_cell_constraints = list(filter(_is_negative_cell_subst, antecedent.constraints))
270                if len(negative_cell_constraints) > 0:
271                    fail_str = (
272                        f'{fail_str}\n\nNegated cell substitutions found (consider using _ => ?_):\n'
273                        + '\n'.join([self.kprint.pretty_print(cc) for cc in negative_cell_constraints])
274                    )
275                return (False, f'Implication check failed, the following is the remaining implication:\n{fail_str}')
276        return (True, '')

278    def cterm_assume_defined(self, cterm: CTerm) -> CTerm:
279        _LOGGER.debug(f'Computing definedness condition for: {cterm}')
280        kast = KApply(KLabel('#Ceil', [GENERATED_TOP_CELL, GENERATED_TOP_CELL]), [cterm.config])
281        kore = self.kprint.kast_to_kore(kast, GENERATED_TOP_CELL)
282        kore_simplified, _logs = self._kore_client.simplify(kore)
283        kast_simplified = self.kprint.kore_to_kast(kore_simplified)
284        _LOGGER.debug(f'Definedness condition computed: {kast_simplified}')
285        return cterm.add_constraint(kast_simplified)

287    def simplify(self, cfg: KCFG, logs: dict[int, tuple[LogEntry, ...]]) -> None:
288        for node in cfg.nodes:
289            _LOGGER.info(f'Simplifying node {self.id}: {shorten_hashes(node.id)}')
290            new_term, next_node_logs = self.cterm_simplify(node.cterm)
291            if new_term != node.cterm:
292                cfg.replace_node(node.id, new_term)
293                if node.id in logs:
294                    logs[node.id] += next_node_logs
295                else:
296                    logs[node.id] = next_node_logs

298    def step(
299        self,
300        cfg: KCFG,
301        node_id: NodeIdLike,
302        logs: dict[int, tuple[LogEntry, ...]],
303        depth: int = 1,
304        module_name: str | None = None,
305    ) -> int:
306        if depth <= 0:
307            raise ValueError(f'Expected positive depth, got: {depth}')
308        node = cfg.node(node_id)
309        successors = list(cfg.successors(node.id))
310        if len(successors) != 0 and type(successors[0]) is KCFG.Split:
311            raise ValueError(f'Cannot take step from split node {self.id}: {shorten_hashes(node.id)}')
312        _LOGGER.info(f'Taking {depth} steps from node {self.id}: {shorten_hashes(node.id)}')
313        exec_res = self.cterm_execute(node.cterm, depth=depth, module_name=module_name)
314        if exec_res.depth != depth:
315            raise ValueError(f'Unable to take {depth} steps from node, got {exec_res.depth} steps {self.id}: {node.id}')
316        if len(exec_res.next_states) > 0:
317            raise ValueError(f'Found branch within {depth} steps {self.id}: {node.id}')
318        new_node = cfg.create_node(exec_res.state)
319        _LOGGER.info(f'Found new node at depth {depth} {self.id}: {shorten_hashes((node.id, new_node.id))}')
320        logs[new_node.id] = exec_res.logs
321        out_edges = cfg.edges(source_id=node.id)
322        if len(out_edges) == 0:
323            cfg.create_edge(node.id, new_node.id, depth=depth)
324        else:
325            edge = out_edges[0]
326            if depth > edge.depth:
327                raise ValueError(
328                    f'Step depth {depth} greater than original edge depth {edge.depth} {self.id}: {shorten_hashes((edge.source.id, edge.target.id))}'
329                )
330            cfg.remove_edge(edge.source.id, edge.target.id)
331            cfg.create_edge(edge.source.id, new_node.id, depth=depth)
332            cfg.create_edge(new_node.id, edge.target.id, depth=(edge.depth - depth))
333        return new_node.id

335    def section_edge(
336        self,
337        cfg: KCFG,
338        source_id: NodeIdLike,
339        target_id: NodeIdLike,
340        logs: dict[int, tuple[LogEntry, ...]],
341        sections: int = 2,
342    ) -> tuple[int, ...]:
343        if sections <= 1:
344            raise ValueError(f'Cannot section an edge less than twice {self.id}: {sections}')
345        edge = single(cfg.edges(source_id=source_id, target_id=target_id))
346        section_depth = int(edge.depth / sections)
347        if section_depth == 0:
348            raise ValueError(f'Too many sections, results in 0-length section {self.id}: {sections}')
349        orig_depth = edge.depth
350        new_depth = section_depth
351        new_nodes = []
352        curr_node_id = edge.source.id
353        while new_depth < orig_depth:
354            _LOGGER.info(f'Taking {section_depth} steps from node {self.id}: {shorten_hashes(curr_node_id)}')
355            curr_node_id = self.step(cfg, curr_node_id, logs, depth=section_depth)
356            new_nodes.append(curr_node_id)
357            new_depth += section_depth
358        return tuple(new_nodes)

380    def check_extendable(self, kcfg_exploration: KCFGExploration, node: KCFG.Node) -> None:
381        kcfg: KCFG = kcfg_exploration.kcfg
382        if not kcfg.is_leaf(node.id):
383            raise ValueError(f'Cannot extend non-leaf node {self.id}: {node.id}')
384        if kcfg.is_stuck(node.id):
385            raise ValueError(f'Cannot extend stuck node {self.id}: {node.id}')
386        if kcfg.is_vacuous(node.id):
387            raise ValueError(f'Cannot extend vacuous node {self.id}: {node.id}')
388        if kcfg_exploration.is_terminal(node.id):
389            raise ValueError(f'Cannot extend terminal node {self.id}: {node.id}')

391    def extend_cterm(
392        self,
393        _cterm: CTerm,
394        *,
395        execute_depth: int | None = None,
396        cut_point_rules: Iterable[str] = (),
397        terminal_rules: Iterable[str] = (),
398        module_name: str | None = None,
399    ) -> ExtendResult:
400        abstract_cterm = self.kcfg_semantics.abstract_node(_cterm)
401        if _cterm != abstract_cterm:
402            return Abstract(abstract_cterm)
403
404        _branches = self.kcfg_semantics.extract_branches(_cterm)
405        branches = []
406        for constraint in _branches:
407            kast = mlAnd(list(_cterm.constraints) + [constraint])
408            kast, _ = self.kast_simplify(kast)
409            if not CTerm._is_bottom(kast):
410                branches.append(constraint)
411        if len(branches) > 1:
412            return Branch(branches, heuristic=True)
413
414        cterm, next_cterms, depth, vacuous, next_node_logs = self.cterm_execute(
415            _cterm,
416            depth=execute_depth,
417            cut_point_rules=cut_point_rules,
418            terminal_rules=terminal_rules,
419            module_name=module_name,
420        )
421
422        # Basic block
423        if depth > 0:
424            return Step(cterm, depth, next_node_logs)
425
426        # Stuck or vacuous
427        if not next_cterms:
428            if vacuous:
429                return Vacuous()
430            return Stuck()
431
432        # Cut rule
433        if len(next_cterms) == 1:
434            return Step(next_cterms[0], 1, next_node_logs, cut=True)
435
436        # Branch
437        assert len(next_cterms) > 1
438        branches = [mlAnd(c for c in s.constraints if c not in cterm.constraints) for s in next_cterms]
439        branch_and = mlAnd(branches)
440        branch_patterns = [
441            mlAnd([mlEqualsTrue(KVariable('B')), mlEqualsTrue(notBool(KVariable('B')))]),
442            mlAnd([mlEqualsTrue(notBool(KVariable('B'))), mlEqualsTrue(KVariable('B'))]),
443            mlAnd([mlEqualsTrue(KVariable('B')), mlEqualsFalse(KVariable('B'))]),
444            mlAnd([mlEqualsFalse(KVariable('B')), mlEqualsTrue(KVariable('B'))]),
445            mlAnd([mlNot(KVariable('B')), KVariable('B')]),
446            mlAnd([KVariable('B'), mlNot(KVariable('B'))]),
447            mlAnd(
448                [
449                    mlEqualsTrue(ltInt(KVariable('I1'), KVariable('I2'))),
450                    mlEqualsTrue(leInt(KVariable('I2'), KVariable('I1'))),
451                ]
452            ),
453            mlAnd(
454                [
455                    mlEqualsTrue(leInt(KVariable('I1'), KVariable('I2'))),
456                    mlEqualsTrue(ltInt(KVariable('I2'), KVariable('I1'))),
457                ]
458            ),
459        ]
460
461        # Split on branch patterns
462        if any(branch_pattern.match(branch_and) for branch_pattern in branch_patterns):
463            return Branch(branches)
464
465        # NDBranch on successor nodes
466        return NDBranch(next_cterms, next_node_logs)

468    def extend_kcfg(
469        self,
470        extend_result: ExtendResult,
471        kcfg: KCFG,
472        node: KCFG.Node,
473        logs: dict[int, tuple[LogEntry, ...]],
474    ) -> None:
475        def log(message: str, *, warning: bool = False) -> None:
476            _LOGGER.log(logging.WARNING if warning else logging.INFO, f'Extend result for {self.id}: {message}')
477
478        def extract_rule_labels(_logs: tuple[LogEntry, ...]) -> list[str]:
479            _rule_lines = []
480            for node_log in _logs:
481                if type(node_log.result) is RewriteSuccess:
482                    if node_log.result.rule_id in self.kprint.definition.sentence_by_unique_id:
483                        sent = self.kprint.definition.sentence_by_unique_id[node_log.result.rule_id]
484                        _rule_lines.append(f'{sent.label}:{sent.source}')
485                    else:
486                        _LOGGER.warning(f'Unknown unique id attached to rule log entry: {node_log}')
487                        _rule_lines.append('UNKNOWN')
488            return _rule_lines
489
490        match extend_result:
491            case Vacuous():
492                kcfg.add_vacuous(node.id)
493                log(f'vacuous node: {node.id}', warning=True)
494
495            case Stuck():
496                kcfg.add_stuck(node.id)
497                log(f'stuck node: {node.id}')
498
499            case Abstract(cterm):
500                new_node = kcfg.create_node(cterm)
501                kcfg.create_cover(node.id, new_node.id)
502                log(f'abstraction node: {node.id} -> {new_node.id}')
503
504            case Step(cterm, depth, next_node_logs, cut):
505                next_node = kcfg.create_node(cterm)
506                logs[next_node.id] = next_node_logs
507                kcfg.create_edge(node.id, next_node.id, depth, rules=extract_rule_labels(next_node_logs))
508                cut_str = 'cut-rule ' if cut else ''
509                log(f'{cut_str}basic block at depth {depth}: {node.id} -> {next_node.id}')
510
511            case Branch(constraints, heuristic):
512                kcfg.split_on_constraints(node.id, constraints)
513                heur_str = ' using heuristics' if heuristic else ''
514                constraint_strs = [self.kprint.pretty_print(bc) for bc in constraints]
515                log(f'{len(constraints)} branches{heur_str}: {node.id} -> {constraint_strs}')
516
517            case NDBranch(cterms, next_node_logs):
