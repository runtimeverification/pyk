15def import_from_file(module_name: str, module_file: str | Path) -> ModuleType:
16    module_file = Path(module_file).resolve()
17    check_file_path(module_file)
18
19    spec = spec_from_file_location(module_name, module_file)
20    if not spec:
21        raise ValueError('Could not create ModuleSpec')
22
23    module = module_from_spec(spec)
24    if not module:
25        raise ValueError('Could not create ModuleType')

27    if not spec.loader:

29
30    spec.loader.exec_module(module)
