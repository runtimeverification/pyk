71    @final
72    def manifest(self) -> dict[str, Any]:
73        source = {}
74        package_path = utils.package_path(self)
75        source_files = [file.resolve() for source in self.source() for file in utils.files_for_path(source)]
76        for source_file in source_files:
77            try:
78                file_id = str(source_file.relative_to(package_path))
79            except ValueError as err:
80                raise ValueError(f'Source file is not within package: {source_file}') from err
81            source[file_id] = utils.timestamp(source_file)
82
83        context = dict(self.context())
84        return {'context': context, 'source': source}
