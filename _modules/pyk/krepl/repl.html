 73    def init_state(self) -> KState:
 74        try:
 75            proc_res = _krun(
 76                input_file=self.program_file,
 77                definition_dir=self.definition_dir,
 78                output=KRunOutput.KORE,
 79                depth=0,
 80            )
 81        except RuntimeError as err:
 82            raise ReplError('Failed to load program') from err
 83
 84        pattern = KoreParser(proc_res.stdout).pattern()
 85        return KState(self.definition_dir, pattern)

135    def do_step(self, args: Namespace) -> None:
136        try:
137            interpreter, state = self._check_state()
138            self._check_steps(args.steps)
139            self.state = interpreter.next_state(state, args.steps)
140        except ReplError as err:
141            self.poutput(err)

152    def _check_state(self) -> tuple[Interpreter, T]:
153        if self.interpreter is None:
154            raise ReplError('No program is loaded')
155        assert self.state is not None
156        return self.interpreter, self.state
157
158    def _check_steps(self, steps: int | None = None) -> None:
159        if steps and steps < 0:
160            raise ReplError('Depth should be non-negative')
