 51class Proof(ABC):
 52    """Should represent a computer proof of a single claim"""

 54    @property
 55    @abstractmethod
 56    def status(self) -> ProofStatus:
 57        """
 58        ProofStatus.PASSED: the claim has been proven
 59        ProofStatus.FAILED: the claim has not been proven, but the proof cannot advance further.
 60        ProofStatus.PENDING: the claim has not yet been proven, but the proof can advance further.
 61        Must not change, except with calls to `prover.commit(self, update)` for some `prover,update`.
 62        If proof.status() is ProofStatus.PENDING, prover.steps(proof) must be nonempty.
 63        If proof.status() is ProofStatus.PASSED, prover.steps(proof) must be empty.
 64        Once proof.status() is ProofStatus.PASSED or ProofStatus.FAILED, it must remain so.

 88def prove_parallel(
 89    proofs: Mapping[str, Proof],
 90    provers: Mapping[str, Prover],
 91    max_workers: int,
 92) -> Iterable[Proof]:
 93    pending: dict[Future[Any], str] = {}
 94    explored: set[tuple[str, ProofStep]] = set()
 95
 96    def submit(proof_id: str, pool: Executor) -> None:
 97        proof = proofs[proof_id]
 98        prover = provers[proof_id]
 99        for step in prover.steps(proof):  # <-- get next steps (represented by e.g. pending nodes, ...)
100            if (proof_id, step) in explored:
101                continue
102            explored.add((proof_id, step))
103            future = pool.submit(step.exec)  # <-- schedule steps for execution
104            pending[future] = proof_id
105
106    with ProcessPoolExecutor(max_workers=max_workers) as pool:
107        for proof_id in proofs:
108            submit(proof_id, pool)
109
110        while pending:
111            done, _ = wait(pending, return_when='FIRST_COMPLETED')
112            future = done.pop()
113
114            proof_id = pending[future]
115            proof = proofs[proof_id]
116            prover = provers[proof_id]
117            try:
118                update = future.result()
119            except CancelledError as err:
120                raise RuntimeError(f'Task was cancelled for proof {proof_id}') from err
121            except TimeoutError as err:
122                raise RuntimeError(
123                    f"Future for proof {proof_id} was not finished executing and timed out. This shouldn't happen since this future was already waited on."
124                ) from err
125            except Exception as err:
126                raise RuntimeError('Exception was raised in ProofStep.exec() for proof {proof_id}.') from err
127
128            prover.commit(proof, update)  # <-- update the proof (can be in-memory, access disk with locking, ...)
129
130            match proof.status:
131                # terminate on first failure, yield partial results, etc.
132                case ProofStatus.FAILED:
133                    ...
134                case ProofStatus.PENDING:
135                    if not list(prover.steps(proof)):
136                        raise ValueError('Prover violated expectation. status is pending with no further steps.')
137                case ProofStatus.PASSED:
138                    if list(prover.steps(proof)):
139                        raise ValueError('Prover violated expectation. status is passed with further steps.')
140
141            submit(proof_id, pool)
142            pending.pop(future)
143    return proofs.values()
