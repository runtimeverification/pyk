 77    def write_proof(self, subproofs: bool = False) -> None:
 78        if not self.proof_dir:
 79            return
 80        proof_path = self.proof_dir / f'{hash_str(self.id)}.json'
 81        if not self.up_to_date:
 82            proof_json = json.dumps(self.dict)
 83            proof_path.write_text(proof_json)
 84            _LOGGER.info(f'Updated proof file {self.id}: {proof_path}')
 85        if subproofs:
 86            for sp in self.subproofs:
 87                sp.write_proof(subproofs=subproofs)

103    @property
104    def up_to_date(self) -> bool:
105        """
106        Check that the proof's representation on disk is up-to-date.
107        """
108        if self.proof_dir is None:
109            raise ValueError(f'Cannot check if proof {self.id} with no proof_dir is up-to-date')
110        proof_path = self.proof_dir / f'{hash_str(id)}.json'
111        if proof_path.exists() and proof_path.is_file():
112            return self.digest == hash_file(proof_path)
113        else:
114            return False

214    @classmethod
215    def read_proof(cls: type[Proof], id: str, proof_dir: Path) -> Proof:
216        # these local imports allow us to call .to_dict() based on the proof type we read from JSON
217        from .implies import EqualityProof, RefutationProof  # noqa
218        from .reachability import APRProof  # noqa
219
220        proof_path = proof_dir / f'{hash_str(id)}.json'
221        if Proof.proof_exists(id, proof_dir):
222            proof_dict = json.loads(proof_path.read_text())
223            proof_type = proof_dict['type']
224            admitted = proof_dict.get('admitted', False)
225            _LOGGER.info(f'Reading {proof_type} from file {id}: {proof_path}')
226            if proof_type in Proof._PROOF_TYPES:
227                return locals()[proof_type].from_dict(proof_dict, proof_dir)
228
229        raise ValueError(f'Could not load Proof from file {id}: {proof_path}')

318    def advance_proof(self, max_iterations: int | None = None, fail_fast: bool = False) -> None:
319        iterations = 0
320        while self.proof.can_progress:
321            if fail_fast and self.proof.failed:
322                _LOGGER.warning(f'Terminating proof early because fail_fast is set: {self.proof.id}')
323                self.proof.failure_info = self.failure_info()
324                return
325            if max_iterations is not None and max_iterations <= iterations:
326                return
327            iterations += 1
328            results = self.step_proof()
329            for result in results:
330                self.proof.commit(result)
331            self.proof.write_proof_data()
332        if self.proof.failed:
333            self.proof.failure_info = self.failure_info()
