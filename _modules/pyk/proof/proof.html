 73    def write_proof(self, subproofs: bool = False) -> None:
 74        if not self.proof_dir:
 75            return
 76        proof_path = self.proof_dir / f'{hash_str(self.id)}.json'
 77        if not self.up_to_date:
 78            proof_json = json.dumps(self.dict)
 79            proof_path.write_text(proof_json)
 80            _LOGGER.info(f'Updated proof file {self.id}: {proof_path}')
 81        if subproofs:
 82            for sp in self.subproofs:
 83                sp.write_proof(subproofs=subproofs)

 99    @property
100    def up_to_date(self) -> bool:
101        """
102        Check that the proof's representation on disk is up-to-date.
103        """
104        if self.proof_dir is None:
105            raise ValueError(f'Cannot check if proof {self.id} with no proof_dir is up-to-date')
106        proof_path = self.proof_dir / f'{hash_str(id)}.json'
107        if proof_path.exists() and proof_path.is_file():
108            return self.digest == hash_file(proof_path)
109        else:
110            return False

112    def read_subproof(self, proof_id: str) -> None:
113        if self.proof_dir is None:
114            raise ValueError(f'Cannot add subproof to the proof {self.id} with no proof_dir')
115        assert self.proof_dir

118        self._subproofs[proof_id] = self.fetch_subproof(proof_id, force_reread=True)

198    @classmethod
199    def read_proof(cls: type[Proof], id: str, proof_dir: Path) -> Proof:
200        # these local imports allow us to call .to_dict() based on the proof type we read from JSON
201        from .equality import EqualityProof, RefutationProof  # noqa
202        from .reachability import APRBMCProof, APRProof  # noqa
203
204        proof_path = proof_dir / f'{hash_str(id)}.json'
205        if Proof.proof_exists(id, proof_dir):
206            proof_dict = json.loads(proof_path.read_text())
207            proof_type = proof_dict['type']
208            admitted = proof_dict.get('admitted', False)
209            _LOGGER.info(f'Reading {proof_type} from file {id}: {proof_path}')
210            if proof_type in Proof._PROOF_TYPES:
211                return locals()[proof_type].from_dict(proof_dict, proof_dir)
212
213        raise ValueError(f'Could not load Proof from file {id}: {proof_path}')
