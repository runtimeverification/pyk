 16def get_rule_by_id(definition: KDefinition, rule_id: str) -> KRule:
 17    """Get a rule from the definition by coverage rule id.
 18
 19    Input:
 20
 21        -   definition: json encoded definition.
 22        -   rule_id: string of unique rule identifier generated by `kompile --coverage`.
 23
 24    Output: json encoded rule which has identifier rule_id.
 25    """
 26
 27    for module in definition.modules:
 28        for sentence in module.sentences:
 29            if type(sentence) is KRule:
 30                if Atts.UNIQUE_ID in sentence.att and sentence.att[Atts.UNIQUE_ID] == rule_id:
 31                    return sentence
 32    raise ValueError(f'Could not find rule with ID: {rule_id}')

 35def strip_coverage_logger(rule: KRule) -> KRule:
 36    body = rule.body
 37    if type(body) is KRewrite:
 38        lhs = body.lhs
 39        rhs = body.rhs
 40        if type(rhs) is KApply and rhs.label.name.startswith('project:'):
 41            rhs_seq = rhs.args[0]
 42            if type(rhs_seq) is KSequence and rhs_seq.arity == 2:
 43                body = KRewrite(lhs, rhs_seq.items[1])
 44    return rule.let(body=body)

 47def translate_coverage(
 48    src_all_rules: Iterable[str],
 49    dst_all_rules: Iterable[str],
 50    dst_definition: KDefinition,
 51    src_rules_list: Iterable[str],
 52) -> list[str]:
 53    """Translate the coverage data from one kompiled definition to another.
 54
 55    Input:
 56
 57        -   src_all_rules: contents of allRules.txt for definition which coverage was generated for.
 58        -   dst_all_rules: contents of allRules.txt for definition which you desire coverage for.
 59        -   dst_definition: JSON encoded definition of dst kompiled definition.
 60        -   src_rules_list: Actual coverage data produced.
 61
 62    Output: list of non-functional rules applied in dst definition translated from src definition.
 63    """
 64
 65    # Load the src_rule_id -> src_source_location rule map from the src kompiled directory
 66    src_rule_map = {}
 67    for line in src_all_rules:
 68        src_rule_hash, src_rule_loc = line.split(' ')
 69        src_rule_loc = src_rule_loc.split('/')[-1]
 70        src_rule_map[src_rule_hash.strip()] = src_rule_loc.strip()
 71
 72    # Load the dst_rule_id -> dst_source_location rule map (and inverts it) from the dst kompiled directory
 73    dst_rule_map = {}
 74    for line in dst_all_rules:
 75        dst_rule_hash, dst_rule_loc = line.split(' ')
 76        dst_rule_loc = dst_rule_loc.split('/')[-1]
 77        dst_rule_map[dst_rule_loc.strip()] = dst_rule_hash.strip()
 78
 79    src_rule_list = [rule_hash.strip() for rule_hash in src_rules_list]
 80
 81    # Filter out non-functional rules from rule map (determining if they are functional via the top symbol in the rule being `<generatedTop>`)
 82    dst_non_function_rules = []
 83    for module in dst_definition.modules:
 84        for sentence in module.sentences:
 85            if type(sentence) is KRule:
 86                body = sentence.body
 87                if (type(body) is KApply and body.label.name == '<generatedTop>') or (
 88                    type(body) is KRewrite and type(body.lhs) is KApply and body.lhs.label.name == '<generatedTop>'
 89                ):
 90                    if Atts.UNIQUE_ID in sentence.att:
 91                        dst_non_function_rules.append(sentence.att[Atts.UNIQUE_ID])
 92
 93    # Convert the src_coverage rules to dst_no_coverage rules via the maps generated above
 94    dst_rule_list = []
 95    for src_rule in src_rule_list:
 96        if src_rule not in src_rule_map:
 97            raise ValueError(f'Could not find rule in src_rule_map: {src_rule}')
 98        src_rule_loc = src_rule_map[src_rule]
 99
100        if src_rule_loc not in dst_rule_map:
101            raise ValueError(f'Could not find rule location in dst_rule_map: {src_rule_loc}')
102        dst_rule = dst_rule_map[src_rule_loc]
103
104        if dst_rule in dst_non_function_rules:
105            dst_rule_list.append(dst_rule)
106
107    return dst_rule_list

110def translate_coverage_from_paths(src_kompiled_dir: str, dst_kompiled_dir: str, src_rules_file: PathLike) -> list[str]:
111    """Translate coverage information given paths to needed files.
112
113    Input:
114
115        -   src_kompiled_dir: Path to kompiled directory of source.
116        -   dst_kompiled_dir: Path to kompiled directory of destination.
117        -   src_rules_file: Path to generated rules coverage file.
118
119    Output: Translated list of rules with non-semantic rules stripped out.
120    """
121    src_all_rules = []
122    with open(src_kompiled_dir + '/allRules.txt') as src_all_rules_file:
123        src_all_rules = [line.strip() for line in src_all_rules_file]
124
125    dst_all_rules = []
126    with open(dst_kompiled_dir + '/allRules.txt') as dst_all_rules_file:
127        dst_all_rules = [line.strip() for line in dst_all_rules_file]
128
129    dst_definition = read_kast_definition(dst_kompiled_dir + '/compiled.json')
130
131    src_rules_list = []
132    with open(src_rules_file) as src_rules:
133        src_rules_list = [line.strip() for line in src_rules]
134
135    return translate_coverage(src_all_rules, dst_all_rules, dst_definition, src_rules_list)
